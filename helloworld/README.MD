# Hello, World

## Running Each file type.


C (hello.c)
	 $ gcc -o bin/hello_c hello.c && ./bin/hello_c
C++ (hello.cc)
	 $ g++ -o bin/hello_cc hello.cc && ./bin/hello_cc
Go (hello.go)
	 $ go run hello.go
Java (hello.java)
	 $ javac hello.java && java hello
Julia (hello.jl)
	 $ julia hello.jl
JavaScript (hello.js)
	 $ node hello.js
PHP (hello.php)
	 $ php hello.php
Perl (hello.pl)
	 $ perl hello.pl
Python (hello.py)
	 $ python hello.py
Ruby (hello.rb)
	 $ ruby hello.rb
Rust (hello.rs)
	 $ rustc hello.rs --crate-name bin/hello_rs && ./hello_rs
Swift (hello.swift)
	 $ swift hello.swift
